import requests
from bs4 import BeautifulSoup

url = "https://thescoop.co/"
r = requests.get(url)
r_html = r.text

soup = BeautifulSoup(r_html, 'html.parser')
a_all = soup.find_all('h2')

#print(a_all)
for h2 in a_all:
    try:
        print("*" , h2.a['title'] , "\n")
    except:
        pass
