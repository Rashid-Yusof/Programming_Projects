import requests
from bs4 import BeautifulSoup

url = "https://thescoop.co/"
r = requests.get(url)
r_html = r.text

soup = BeautifulSoup(r_html, 'html.parser')
a_all = soup.find_all('h2')

#print(a_all)
#lines = []
with open('Scoop.txt' , 'w') as open_file:
    for h2 in a_all:
        try:
            open_file.write("* " + h2.a['title'] + "\n")
            #print("*" , h2.a['title'] , "\n")
            #lines.append("* " + h2.a['title'] + "\n")
        except:
            pass
