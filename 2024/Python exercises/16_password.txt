import string 
import random

def passwordCreator():
    strength = int(input("How strong do you want the password? 0 for weak, 1 for mid\n"))
    if strength == 0:
        return random.choice(["sex", "porn", "rui", "indonesian muusangg"])
    else:
        length = int(input("How long do you want your password? "))
        
        password = ""
        while length != 0:
            choice = random.randint(0, 2)
            if choice == 0:
                password += random.choice(string.ascii_letters)
            elif choice == 1:
                password += str(random.randint(0, 9))
            else:
                password += random.choice("!@#$%^&*()")
            length -= 1
        return password
